function doPost(e) { const SHEET_NAME = "Pengawas Kelas"; const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(["Timestamp","Nama","NIM","Prodi", "No. HP","Dosen","Tanggal","Waktu", "Keterangan","Bank","No. Rekening", "Nama Pemilik Rek.","Status"]); sheet.setFrozenRows(1); } const data = JSON.parse(e.postData.contents); sheet.appendRow([data.timestamp,data.nama,data.nim, data.prodi,data.noHp,data.dosen,data.tanggal, data.waktu,data.keterangan,data.namaBank, data.noRekening,data.namaPemilik,"Belum Diverifikasi"]); return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); }
